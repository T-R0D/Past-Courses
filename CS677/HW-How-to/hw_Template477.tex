	return math.cos(x) - x
def fPrime(x):
	return -math.sin(x) - 1

def newtonRec(a,eps = 10**-5):
	A = a - f(a) / fPrime(a)
	if abs(A - a) < eps:
		return A
	else:
		return newtonRec(A,eps)
